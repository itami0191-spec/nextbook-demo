const MAX_REASON_CHARS = 300;

function trimReasonToMax(text, max = MAX_REASON_CHARS) {
  const arr = Array.from(text);
  if (arr.length <= max) return text;
  if (max <= 1) return "…";
  return arr.slice(0, max - 1).join("") + "…";
}

function buildCompactReason(candidate, dimensionRanking, matchedBooks, traits) {
  const pile = joinedMatchedTitles(matchedBooks, 3);
  const tags = tagLine(candidate, 3);
  const d0 = dimensionRanking[0];
  const d1 = dimensionRanking[1];
  const hints = pickHintsFromLinks(candidate);
  const traitsShort = traits.slice(0, 3).join("、");
  const matchedAuthors = new Set(matchedBooks.map(b => b.author));
  const sameAuthor = matchedAuthors.has(candidate.author);

  const intro = `《${candidate.title}》（${candidate.year}），带着「${tags}」的味道；和你这次勾的「${pile}」并排读，不会太突兀。`;

  const key = d0?.key || "themes";
  let mid = "";
  switch (key) {
    case "lineage":
      mid = hints[0]
        ? `谱系上常有人把它和「${hints[0]}」拴在一根绳上讲。`
        : "在文学史纵轴上，它更像一块承前启后的小月台，适合踩一脚换车。";
      break;
    case "contemporary":
      mid = `同时代语境这条线看，${candidate.year} 年前后的心事，会在字里行间旧得很具体。`;
      break;
    case "social":
      mid = "作家交游这一侧，它的背后是一张可走动的小地图：名字连着名字。";
      break;
    case "authorOther":
      mid = sameAuthor
        ? "同作者补读的话，这本往往是作者把旧执念又拧了一次的那下。"
        : "顺着同作者作品摸，它常负责把作者的固执摊开给你看。";
      break;
    case "themes":
      mid = traitsShort
        ? `主题与标签上，你写的「${traitsShort}」和它算对上了眼神。`
        : "主题一侧，它给你的更像问句集合，而不是现成的答案。";
      break;
    case "custom":
    default:
      mid = traitsShort
        ? `自定义特质一侧，「${traitsShort}」不必对号入座，但气味会撞上几步。`
        : "自定义关键词留空时，它主要靠谱系与标签暗地里指路。";
      break;
  }

  const dimClose = d0 && d1 ? `「${d0.label}」和「${d1.label}」两处都亮灯，值得插队进待读。` : "";

  const full = `${intro}${mid}${dimClose}`;
  return [trimReasonToMax(full)];
}
